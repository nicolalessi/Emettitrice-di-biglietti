CREATE SCHEMA `emettitrici` ;

USE `emettitrici`;

CREATE TABLE `emettitrici`.`users` (
  `USERNAME` VARCHAR(45) NOT NULL,
  `NAME` VARCHAR(45) NOT NULL,
  `SURNAME` VARCHAR(45) NOT NULL,
  `CF` VARCHAR(45) NOT NULL,
  `PSW` VARCHAR(45) NOT NULL,
  `EMAIL` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`USERNAME`));
  
CREATE TABLE `emettitrici`.`sales` (
  `SERIALCODE` INT NOT NULL,
  `TYPE` VARCHAR(45) NULL,
  `USER` VARCHAR(45) NOT NULL,
  `SELL_DATE` VARCHAR(45) NOT NULL,
  `SELLER_IP` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`SERIALCODE`));
  
CREATE TABLE `emettitrici`.`informations` (
  `KEY` VARCHAR(45) NOT NULL,
  `VALUE` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`KEY`));
  
CREATE TABLE `emettitrici`.`fines` (
  `ID` VARCHAR(45) NOT NULL,
  `CF` VARCHAR(45) NOT NULL,
  `AMOUNT` DOUBLE NOT NULL,
  `COLLECTOR_USERNAME` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`ID`));
  
CREATE TABLE `emettitrici`.`collectors` (
  `USERNAME` VARCHAR(45) NOT NULL,
  `NAME` VARCHAR(45) NOT NULL,
  `SURNAME` VARCHAR(45) NOT NULL,
  `CF` VARCHAR(45) NOT NULL,
  `PSW` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`USERNAME`));
